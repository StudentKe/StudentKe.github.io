//顺序栈的存储类型描述

define MaxSize 50

typedef struct{
ElemType data[MaxSize];
int top;
}SqStack;

//栈初始化
void InitStack(SqStack &S){
S.top = -1;
}
//判断栈空
bool StackEmpty(SqStack S){
if(S.top == -1)
return true;
else
return false;
}
//进栈
bool Push(SqStack &S,ElemType x){
if(S.top == MaxSize-1)
return false; //栈满
S.data[++top] = x; //指针先+1 再入栈
return true;
}
//出栈
bool Pop(SqStack &S,ElemType &x){
if(S.top == -1)
return false; //栈空
x = S.data[top—]; //先出栈,再-1
return true;
}
//读栈顶元素
bool GetTop(SqStack S,ElemType &x){
if(S.top == -1)
return false;
x = S.data[top];
return true;
}

//二叉树的非递归后序遍历
//思路二
//对于任意一个节点：
// 如果是叶子节点，可以直接访问
// 如果是非叶子节点，但已经访问过左右孩子，可以访问
//此思路需要：
// 设置一个pre指针指示上一个访问过的节点
// 设置一个cur指针指示当前访问的节点
//当节点是非叶子节点时，将右左节点依次入栈，
//可以保证右节点存在的情况下一定在左节点之后访问，
//即若cur -> lchild == pre(没有右结点)
//或cur -> rchild == pre(访问右结点完毕)即可访问当前节点。
void PostOrder2(BiTree T){
InitStack(S);
BiTree p = T;
Push(S,p);
BiTNode pre = NULL;
BiTNode cur;
while (!isEmpty(S)) {
cur = GetTop(S,p);
if( cur->lchild == null && cur->rchild == null ||
(pre!=null) && (cur->lchild = pre || cur->rchild = pre)
){
Pop(S,p);
visit(p);
pre = cur;
}else{
if(cur->lchild)
Push(S,cur->lchild);
if(cur->rchild)
Push(S,cur->rchild);
}
}
}

//线索二叉树的存储结构
typedef struct ThreadNode{
ElemType data;
struct ThreadNode lchild,rchild;
int ltag,rtag;
}ThreadNode,*ThreadNode;

//中序遍历线索化递归算法
//递归操作中 因为传入的参数为引用参数，每层递归都会修改指针，
//为了避免下层递归修改上层递归的当前节点p指针 ，每次传参直接传入的是p->child 而不是 p = ??
void InThread(ThreadNode &p,ThreadNode &pre){
if(p!=null){
ThreadNode(p->lchild,pre);
if(p->lchild == NULL){
p->lchild = pre;
p->ltag = 1;
}
if(pre!=NULL && pre->rchild == NULL){
pre->rchild = p;
pre->rtag = 1;
}
pre = p;
InThread(p->rtag,pre);
}
}

void CreateInThread(ThreadTree T){
ThreadTree pre = NULL;
if(T!=NULL){
InThread(T,pre);
// 处理最后一个节点
pre->rchild = NULL;
pre->rtag = 1;
}
}

//求中序线索二叉树的第一个节点
ThreadNode Firstnode(ThreadNode p){
while (p->lchild == 0) {
p = p->lchild;

    return p;
}

}

//求p节点在中序遍历下的后继节点
p在中序遍历序列中 右子树的最左侧节点即为后继节点
ThreadNode Nextnode(ThreadNode p){
if(p->ltag == 0) return Firstnode(p->rchild);
else return p->rchild;
}
